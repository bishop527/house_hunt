from google.cloud import monitoring_v3
import datetime

# Initialize client
client = monitoring_v3.MetricServiceClient()
project_name = f"projects/{YOUR_PROJECT_ID}"

# Set time interval (e.g., current month)
now = datetime.datetime.utcnow()
start_of_month = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0)

interval = monitoring_v3.TimeInterval({
    "end_time": {"seconds": int(now.timestamp())},
    "start_time": {"seconds": int(start_of_month.timestamp())}
})

# Query for Distance Matrix API usage
results = client.list_time_series(
    request={
        "name": project_name,
        "filter": 'metric.type="serviceruntime.googleapis.com/api/request_count" AND resource.labels.service="distancematrix-backend.googleapis.com"',
        "interval": interval,
        "view": monitoring_v3.ListTimeSeriesRequest.TimeSeriesView.FULL
    }
)

# Sum up total requests (elements)
total_elements = 0
for result in results:
    for point in result.points:
        total_elements += point.value.int64_value

print(f"Total Distance Matrix elements this month: {total_elements}")
print(f"Free tier remaining: {max(0, 10000 - total_elements)}")
